write an Example program for Method OverLoading?
public class Employees { void getSalary(int basic, float hike, float pf) { float salary = (basic + ((basic + hike) / 100) - ((basic * pf) / 100)); System.out.println("salary " + salary); } void getSalary(int basic, float hike, float pf, int tax, int bonus) { float salary = (basic + ((basic + hike) / 100) - ((basic * pf) / 100) + tax + bonus); System.out.println("salary " + salary); } public static void main(String args[]) { Employees employees = new Employees(); employees.getSalary(20000, 25.0f, 12.5f); employees.getSalary(20000, 25.0f, 12.5f, 2000, 5000); } }
