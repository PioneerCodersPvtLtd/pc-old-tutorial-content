Write example program on methods?
public class MethodEx1 { static int totalLength(String a, String b) { // Add up lengths of two strings. return a.length() + b.length(); } static int averageLength(String a, String b) { // Divide total length by 2. return totalLength(a, b) / 2; } public static void main(String[] args) { // Call methods. int total = totalLength("Pioneer", "Coders"); int average = averageLength("Pioneer", "Coders"); System.out.println("total is "+total); System.out.println("average is "+average); } }
