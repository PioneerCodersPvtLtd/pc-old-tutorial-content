class Aeroplane { public int wings; public int speed; public int accelaration; //Constructor public Aeroplane (int wings, int speed, int accelaration) { this.wings = wings; this.speed = speed; this.accelaration = accelaration; } void display() { System.out.println("Aeroplane has "+wings + " wings and going with speed of " +speed+" and accelaration of "+accelaration +"km/sec"); } public static void main(String[] args){ Aeroplane s1 = new Aeroplane(2,90,180); s1.display(); } }
