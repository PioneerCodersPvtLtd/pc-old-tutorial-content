public class Check { static void myMethod(int testnum) throws Exception { System.out.println("method started"); if (testnum == 12) throw new Exception(); System.out.println("method ended"); return; } public static void main(String args[]) { int testnum = 12; try { System.out.println("try block begin"); myMethod(testnum); System.out.println("try block ends"); } catch (Exception e) { System.out.println("An Exception"); } finally { System.out.println("finally"); } System.out.println("out of try-catch block"); } }
