Write a program using throws
public class FirstThrowsExample { public static void main(String[] args){ try{ M m = new M(); m.method(); } catch(Exception e) { System.out.println("Exception Handled"); } System.out.println("No error in the program"); } } class M { void method() throws Exception { throw new Exception("Error occured in the program"); } }
